#include <stdio.h>
int main()
{
int n, i, j, swap; 
printf("Enter number of elements\n");
scanf("%d", &n); 
int array[n];
printf("Enter %d Numbers:\n", n); 
for(i = 0; i < n; i++){
    
scanf("%d", &array[i]);

} 
i = 0;
while(i < n - 1)
{
for(j = 0 ; j < n-i-1; j++)
{
if(array[j] > array[j+1]) 
{
swap=array[j];
array[j]=array[j+1];
array[j+1]=swap;
}
}
i=i+1;
} 
printf("Sorted Array:\n"); 
for(i = 0; i < n; i++)
printf("%d\n", array[i]);
return 0;
}
